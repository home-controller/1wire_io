280bool writeDS2413(OneWire &o, byte addr[], byte state)
281{
282 uint8_t ack = 0;
283 /* Top six bits must '1' */
284 state |= 0xFC;
285 // state = state xor B11;// Invert IO bits as 0 = 0 voltage and as IO output is an n mosfet 0v is = on;
286 o.reset();
287 o.select(addr);
288 o.write(DS2413_ACCESS_WRITE);
289 o.write(state xor B11); // Invert IO bits as 0 = 0 voltage and as IO output is an n mosfet 0v is = on;
290 o.write(state xor B11111100); /* Invert data and resend */
291 ack = o.read(); /* 0xAA=success, 0xFF=failure */
292 o.reset(); // Stop chip sending. At least in some places the chip read continuously sends on the bus until reset.
293 Serial.print(F("ack:"));
294 Serial.print(ack, HEX);
295 if (ack == DS2413_ACK_SUCCESS)
296 { // 0xAA
297 if ((QuiteMode) != 1)
298 {
299 Serial.print(F("Set success: B"));
300 Serial.print(state, BIN);
301 }
302 }
303 else
304 {
305 Serial.print(F(" Error writing to 1wire, 0xAA=success, 0xFF=failure: 0x"));
306 }
307 return (ack == DS2413_ACK_SUCCESS ? true : false);
308}
