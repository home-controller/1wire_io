733bool writeDS2413( OneWire& o, byte addr[ ], byte state ) {
734 uint8_t ack = 0;
735 /* Top six bits must '1' */
736 state |= 0xFC;
737 // state = state xor B11;// Invert IO bits as 0 = 0 voltage and as IO output is an n mosfet 0v is = on;
738 o.reset();
739 o.select( addr );
740 o.write( DS2413_ACCESS_WRITE );
741 o.write( state xor B11 ); // Invert IO bits as 0 = 0 voltage and as IO output is an n mosfet 0v is = on;
742 o.write( state xor B11111100 ); /* Invert data and resend */
743 ack = o.read(); /* 0xAA=success, 0xFF=failure */
744 o.reset(); // Stop chip sending. At least in some places the chip read continuously sends on the bus until reset.
745 Serial.print( F( "ack:" ) ); Serial.print( ack, HEX );
746 if ( ack == DS2413_ACK_SUCCESS ) {//0xAA
747 if ( !quiteMode ) {
748 Serial.print( F( "Set success: B" ) );
749 Serial.print( state, BIN );
750 }
751 } else {
752 Serial.print( F( " Error writing to 1wire, 0xAA=success, 0xFF=failure: 0x" ) );
753 }
754 return ( ack == DS2413_ACK_SUCCESS ? true : false );
755}
